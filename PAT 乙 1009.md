void printreversely1() {

	std::string s;
	std::getline(std::cin, s);  // read the whole line
	int len = s.length();
	int p = len;
	for (int i = len - 1; i >= 0; i--) {       
		if (s[i] == ' ') {   
			for (int j = i + 1; j <= p - 1; j++) {
				std::cout << s[j];
			}
			std::cout << " ";
			p = i;
		}
	}

	for (int i = 0; i <= p - 1; i++) {
		std::cout << s[i];
	}
}
